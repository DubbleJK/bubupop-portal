# 발주서 발행기 (purchase-order-app) 커밋 후 푸시

# 1) 발주서 앱 폴더로 이동
cd C:\Users\USER\purchase-order-app

# 2) 변경 사항 확인 (선택)
git status

# 3) 변경 파일 스테이징
git add .

# 4) 커밋 (메시지는 수정해서 사용)
git commit -m "발주서 발행기 업데이트"

# 5) GitHub 푸시 (Vercel 자동 배포)
git push origin main
